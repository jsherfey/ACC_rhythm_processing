# A: A-current
gA=1; EA=-80;
a_IC=0; b_IC=1; IC_noise=0;

ainf(V)=1./(1+exp(-(V+60)/8.5))
atau(V)=.37+1./(exp((V+46)/5)+exp(-(V+238)/37.5))
binf(V)=1./(1+exp((V+78)/6))
btau(V)=19+1./(exp((V+46)/5)+exp((V+238)/(-37.5)))
a' = (ainf(V)-a)./atau(V)
b' = (binf(V)-b)./btau(V)
a(0)=a_IC+IC_noise.*rand(Npop,1)
b(0)=b_IC+IC_noise.*rand(Npop,1)
IA(V,a,b)=gA.*a.^4.*b.*(V-EA)
current => -IA(V,a,b)
