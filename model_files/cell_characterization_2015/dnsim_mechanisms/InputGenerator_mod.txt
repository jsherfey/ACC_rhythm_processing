% InputGenerator.txt
amp=1; % amplitude
onset=0;
offset=inf;
inputtype=1;
%timelimits=1000; % [beg end], if scalar, InputGenerator() assumes [0 end]

% Type-specific parameters

% type 1 (tonic)

% type 2 (sin)
f0=10; % oscillation frequency

% type 3 (ZAP)
fmin=1; % start frequency
fmax=40; % final frequency
deltaf=1; % step between frequencies
Ncycles=3; % number of cycles at each frequency

% type 4...
% ...

% type 7 (Poisson; Ben's method)
sharedfraction=0;
Ninputs=10;
lambda=50;%2;     % Hz, Poisson spike rate
tauD=2;
tauR=.5;

%I=InputGenerator(timelimits,dt,inputtype,Npop,amp,onset,offset,f0,fmin,fmax,deltaf,Ncycles,sharedfraction);
I=InputGenerator('timelimits',timelimits,'dt',dt,'inputtype',inputtype,'Npop',Npop,'amp',amp,'onset',onset,'offset',offset,'f0',f0,'fmin',fmin,'fmax',fmax,'deltaf',deltaf,'Ncycles',Ncycles,'sharedfraction',sharedfraction,'Ninputs',Ninputs,'tauD',tauD,'tauR',tauR,'lambda',lambda);

input(t) = I(floor(1+(t/dt)));

current => input(t)
